% Simple emoji support for XeLaTeX
\usepackage{fontspec}

% Set main fonts with good Unicode coverage
\setmainfont{DejaVu Sans}
\setsansfont{DejaVu Sans}
\setmonofont{DejaVu Sans Mono}

% Enable Unicode support
\usepackage[utf8]{inputenc}
\usepackage{xunicode}
\usepackage{xltxtra}

% This allows basic emoji rendering through Unicode
% Most emojis will render as black and white symbols
% but they will be visible in the document
